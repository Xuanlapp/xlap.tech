FULL LENH POWERSHELL SETUP XLAP2
================================

# 1. Di vao xlap2
Set-Location -LiteralPath "I:\XLAP\Web\VPS\xlap2"

# 2. Copy .env tu xlap sang xlap2
Copy-Item -LiteralPath "I:\XLAP\Web\VPS\xlap\.env" -Destination "I:\XLAP\Web\VPS\xlap2\.env" -Force

# 3. Copy cache/model neu can dung tach nen giong xlap
if (Test-Path -LiteralPath "I:\XLAP\Web\VPS\xlap\.transformers-cache") {
    Copy-Item -LiteralPath "I:\XLAP\Web\VPS\xlap\.transformers-cache" -Destination "I:\XLAP\Web\VPS\xlap2" -Recurse -Force
}

# 4. Copy storage public de giu anh generated, PSD mockup, va file /storage/... cu
if (Test-Path -LiteralPath "I:\XLAP\Web\VPS\xlap\storage\app\public") {
    Copy-Item -LiteralPath "I:\XLAP\Web\VPS\xlap\storage\app\public" -Destination "I:\XLAP\Web\VPS\xlap2\storage\app" -Recurse -Force
}

# 5. Copy Google Drive service account neu xlap dang co file fallback nay
if (Test-Path -LiteralPath "I:\XLAP\Web\VPS\xlap\storage\app\google-drive.json") {
    Copy-Item -LiteralPath "I:\XLAP\Web\VPS\xlap\storage\app\google-drive.json" -Destination "I:\XLAP\Web\VPS\xlap2\storage\app\google-drive.json" -Force
}

# 6. Cai PHP dependencies
php composer.phar install

# 7. Cai Node dependencies cho Vite va PSD renderer
npm ci

# 8. Clear cache truoc khi tao link/build
php artisan optimize:clear
php artisan view:clear

# 9. Tao storage link
php artisan storage:link

# Neu buoc tren bao "public/storage already exists", chay 2 lenh nay roi chay lai storage:link:
# Remove-Item -LiteralPath "I:\XLAP\Web\VPS\xlap2\public\storage" -Force
# php artisan storage:link

# 10. Chay migrate database
php artisan migrate

# 11. Kiem tra PSD renderer mockup tu chon
node --check scripts/psd-renderer/render.js

# 12. Build frontend assets
npm run build

# 13. Kiem tra app
php artisan about
php artisan route:list

# 14. Clear lai cache sau khi kiem tra/sua Cloudflare, Drive, PSD, tich hop ngoai
php artisan optimize:clear
php artisan config:clear

# 15. Test command upload approved images len Google Drive
# Chi chay lenh nay khi da cau hinh Google Drive dung va muon upload that.
# Lenh nay co the thay DB image columns bang Google Drive URL va xoa file local da upload.
# php artisan offorest:upload-approved-images-to-drive

# 16. Test schedule neu can
# php artisan schedule:run

# 17. Chay web local tren port 8002
php artisan serve --host=127.0.0.1 --port=8002

CLOUDFLARE SECURITY CHECKLIST TREN DASHBOARD
============================================

# 1. DNS
# - Neu xlap2 dung domain rieng, tao A/CNAME tro ve server.
# - Bat proxy mau cam neu muon qua Cloudflare WAF/CDN.
# - APP_URL trong .env phai khop domain that.

# 2. SSL/TLS
# - Nen dung Full strict neu server co SSL hop le.
# - Bat Always Use HTTPS.
# - Bat Automatic HTTPS Rewrites neu can.

# 3. Turnstile
# - Tao Turnstile site trong Cloudflare.
# - Add domain cua xlap2 vao allowed domains.
# - Copy Site Key vao TURNSTILE_SITE_KEY.
# - Copy Secret Key vao TURNSTILE_SECRET_KEY.
# - Sau khi sua .env:
php artisan optimize:clear
php artisan config:clear


LENH XU LY BAO MAT DANG NHAP CLOUDFLARE TURNSTILE
=================================================

# Loi thuong gap tren man hinh login:
# - "Vui long hoan tat xac minh bao mat."
# - Widget Cloudflare bao "Khong the ket noi voi trang web"
#
# Nguyen nhan thuong gap:
# - Copy .env tu xlap sang xlap2 nhung Turnstile key chi cho phep domain cu.
# - Dang chay local bang 127.0.0.1:8002 hoac xlap2.test nhung domain nay chua add trong Cloudflare Turnstile.
# - TURNSTILE_SITE_KEY/SECRET_KEY sai cap voi nhau.
# - Trinh duyet/server bi chan ket noi den challenges.cloudflare.com.

# CACH 1: Tat Turnstile khi test local xlap2
# Chay trong PowerShell:
Set-Location -LiteralPath "I:\XLAP\Web\VPS\xlap2"
(Get-Content -LiteralPath ".env") `
    -replace '^TURNSTILE_ENABLED=.*', 'TURNSTILE_ENABLED=false' |
    Set-Content -LiteralPath ".env"
php artisan optimize:clear
php artisan config:clear

# Sau do reload trang login va dang nhap lai.

# CACH 2: Bat Turnstile cho domain xlap2 dung cach
# Buoc lam tren Cloudflare Dashboard:
# 1. Vao Cloudflare Dashboard -> Turnstile.
# 2. Tao widget moi hoac sua widget cu.
# 3. Them domain dang chay xlap2 vao allowed domains, vi du:
#    xlap2.test
#    localhost
#    127.0.0.1
#    ten-domain-production-cua-xlap2.com
# 4. Copy Site Key va Secret Key moi.
#
# Sau do sua .env bang PowerShell:
Set-Location -LiteralPath "I:\XLAP\Web\VPS\xlap2"
(Get-Content -LiteralPath ".env") `
    -replace '^TURNSTILE_ENABLED=.*', 'TURNSTILE_ENABLED=true' `
    -replace '^TURNSTILE_SITE_KEY=.*', 'TURNSTILE_SITE_KEY=PASTE_SITE_KEY_HERE' `
    -replace '^TURNSTILE_SECRET_KEY=.*', 'TURNSTILE_SECRET_KEY=PASTE_SECRET_KEY_HERE' |
    Set-Content -LiteralPath ".env"
php artisan optimize:clear
php artisan config:clear

# Kiem tra bien Turnstile app dang doc:
php artisan tinker
# Trong tinker chay:
# config('services.turnstile.enabled');
# config('services.turnstile.site_key');
# filled(config('services.turnstile.secret_key'));
# exit

# CACH 3: Neu muon sua thu cong trong .env
